% ============================================================
% 1. THE PROBLEM  [JASMIN]
% ============================================================
\begin{frame}{Testing only tries \emph{some} inputs}
  \begin{columns}[T]
    \column{0.52\linewidth}
      \deflab{Testing}{run it on a few inputs, watch what happens}
      \begin{itemize}
        \item just 8 bytes of input $=\ 2^{64}$ possibilities
        \item more than stars in the sky --- you try a handful
      \end{itemize}
      \vspace{0.3cm}
      \deflab{Verification}{reason about \emph{every} input at once}
      \begin{itemize}
        \item treat the input as algebra, not as guesses
        \item ``can it crash, ever, within $k$ steps?''
      \end{itemize}
    \column{0.44\linewidth}
      \centering
      \vspace{0.4cm}
      {\Large $2^{64}$}\\[0.2em]
      {\footnotesize possible inputs}\\[0.8cm]
      {\Large vs.}\\[0.8cm]
      {\large \textbf{1 question}}\\[0.2em]
      {\footnotesize answered for all of them}
  \end{columns}
\end{frame}
\note{[JASMIN --- 1:00]\\
``The normal way to find bugs is testing: you run a program on some inputs
and see if it breaks. But a program reading just eight bytes has more
possible inputs than there are stars in the sky --- and you only ever try a
handful. So testing can't tell you a program is \emph{safe}; only that it
didn't break the times you looked.\\[2pt]
There's a stronger idea: \textbf{verification}. Instead of trying inputs one
by one, you treat the unknown input like algebra --- like solving x plus
three equals seven without guessing every number --- and you ask one
question that covers \emph{all} inputs: can this program reach a bad state,
ever, within k steps? If yes, you even get the exact input that does it.''}

  \begin{tabular}{lrr}

    \bottomrule
  \end{tabular}
  \end{center}

  \begin{block}{The supervisor's reaction: ``don't trust that --- \emph{count}''}
    One repeated question: \emph{``have I built this piece before?''}\\[2pt]
    C walks a list: \textbf{11{,}695{,}232{,}963} comparisons.\hfill
    Rust: one hash lookup --- \textbf{159{,}018}.
  \end{block}
\end{frame}
\note{[JASMIN --- 1:30]\\
``So we rebuilt rotor in Rust. And it ran about a thousand times faster ---
selfie's model dropped from \emph{139 seconds} to under a tenth of a second,
and from 428 megabytes to twenty.\\[2pt]
When we showed Professor Kirsch, he didn't celebrate --- he got
\emph{suspicious}. `Maybe it's faster because it's doing less. Don't trust the
number --- count.' So we did exactly what he taught: we found the one
repeated operation both tools share --- the question `have I already built
this piece of the model before?' --- and put a counter on it in both.\\[2pt]
The C tool answers that question by walking a list of everything it built so
far. Eleven-point-seven \emph{billion} comparisons. Our Rust version asks the
same question with a hash table --- one lookup each, 159 thousand total. Same
question. Same answer. A list versus an index. \emph{That}'s the whole
thousand-x --- nothing skipped.''}

% 5. SYMBOLIC ARGV -- THE GAP  [DANIEL]
% ============================================================
\begin{frame}{A whole class of bugs was invisible}
  \begin{columns}[T]
    \column{0.54\linewidth}
      \deflab{Before}{the solver could only explore what you type at the
                      keyboard (stdin)}
      \begin{itemize}
        \item command-line arguments were \emph{frozen}
        \item a bug needing \texttt{./prog SECRET} was unreachable
      \end{itemize}
      \vspace{0.25cm}
      \deflab{Our addition: symbolic argv}{leave the argument bytes
              \emph{open} for the solver to choose}
  \column{0.42\linewidth}
      \centering
      \vspace{0.3cm}
      \fbox{\parbox{0.92\linewidth}{\centering\footnotesize
        the paper's own words:\\[3pt]
        \textit{``console arguments as symbolic input is \textbf{future
        work}''}}}\\[0.4cm]
      {\footnotesize We built the future-work item.}
  \end{columns}
\end{frame}
\note{[DANIEL --- 1:00]\\
``Thanks. So here's a gap we found. The solver could only explore input typed
at the keyboard --- standard input. But programs also get input from
\emph{command-line arguments}: the words after the program's name. In the old
models, those were frozen. So any bug that only happens for a specific
argument --- like `./program CRASH' --- was \emph{structurally invisible}. The
solver wasn't failing to find it; it wasn't allowed to look.\\[2pt]
We fixed that. We leave the argument bytes \emph{open} --- blanks the solver
fills in. And this wasn't our idea to invent: the original rotor paper lists
`console arguments as symbolic input' as \emph{future work}. We built their
future-work item.''}

% 6. THE X / Y BUG  [DANIEL]
% ============================================================
\begin{frame}{The solver finds the secret by itself}
  \begin{center}
  \texttt{\small if (argv[1][0]=='X' \&\& argv[2][0]=='Y') \textcolor{red}{crash();}}
  \end{center}
  \vspace{0.2cm}
  \begin{itemize}
    \item two specific letters, in \emph{two different} arguments at once
    \item impossible to reach through stdin --- the program never reads it
    \item solver's answer, in seconds: \keylab{argv[1][0]=X}, \keylab{argv[2][0]=Y}
  \end{itemize}
  \vspace{0.15cm}
  \centering\footnotesize Five such programs --- five bugs found, exact bytes,
  none reachable the old way.
\end{frame}
\note{[DANIEL --- 1:00]\\
``Let me make it concrete. This program crashes only if the first letter of
argument one is X \emph{and} the first letter of argument two is Y --- two
specific characters, in two \emph{different} arguments, at the same time. No
amount of keyboard input could ever trigger it; the program never reads the
keyboard.\\[2pt]
We hand the model to the solver, and in seconds it comes back with: argument
one starts with X, argument two starts with Y. It \emph{derived the secret}
--- by logic, not guessing. We wrote five of these traps; the solver cracked
all five, every time with the exact bytes.''}

% ---- Q&A 1: the counting numbers ----
\begin{frame}{Q\&A: the counting, in full}
  \centering\footnotesize
  \begin{tabular}{lrr}
    \toprule
     & C rotor & Rust rotor\\
    \midrule
    node creations            & 3{,}171{,}632       & 159{,}018\\
    dedup question asked       & 9{,}976             & 159{,}018\\
    hits (already present)     & 6{,}021             & 48{,}114\\
    \textbf{basic operations}  & \textbf{11{,}695{,}232{,}963} & \textbf{159{,}018}\\
    \bottomrule
  \end{tabular}\\[0.4em]
  self-check: creations $-$ hits $=$ tool's own reported line count, exactly.
\end{frame}
\note{[ANSWER --- ``how exactly did you measure the speed-up?'']\\
``We instrumented the dedup question in both tools with plain integer
counters, the way the professor counts everything in selfie. C asks it 10
thousand times but each answer scans a million-entry list --- 11.7 billion
comparisons total. Rust asks it every time, but each is one hash probe ---
159 thousand. And the counters are self-consistent: creations minus reuses
equals each tool's own reported line count, to the integer. That's how we
know we counted the right thing.''}

% ---- Q&A 2: why still faster than the optimized C ----
\begin{frame}{Q\&A: Rust vs.\ the \emph{optimized} C}
  \centering\footnotesize
  \begin{tabular}{lrrr}
    \toprule
    selfie & baseline C & C + hash & Rust\\
    \midrule
    time   & 139 s   & 1.5 s   & $<$0.1 s\\
    memory & 428 MB  & 485 MB  & 20 MB\\
    lines created & 3.17 M & 3.17 M & 159 k\\
    \bottomrule
  \end{tabular}\\[0.4em]
  Hash fixed the \emph{lookup}. The rest is \emph{when} you build:
  Rust never creates the duplicates; C still does, then prunes.
\end{frame}
\note{[ANSWER --- ``if the hash is the whole story, why is Rust still faster
than your hashed C?'']\\
``Great question --- two optimizations, not one. The hash fixed the
\emph{lookup cost}. But the C still \emph{creates} 3.17 million line records
and prunes later --- create-then-filter. Rust checks before allocating, so
duplicates never exist --- filter-at-creation. That second difference is the
memory story and the last factor of speed. We even built that into the C too;
it stays equivalent, 36/36.''}

% ---- Q&A 3: the dedup-off experiment + the crash ----
\begin{frame}{Q\&A: ``disable the duplicate check in both''}
  \begin{itemize}\setlength\itemsep{0.4em}
    \item \keylab{C rotor, reuse off:} \textbf{crashes} --- \texttt{ite then
          sort mismatch}. Line reuse is load-bearing, not just speed.
    \item \keylab{Rust, \flag{--no-cse}:} runs fine; model grows 1.4$\times$,
          \emph{same} solver verdict. Dedup is cosmetic, never semantic.
    \item We sent a complete, reproducible crash report + a fix.
  \end{itemize}
\end{frame}
\note{[ANSWER --- the professor's own experiment]\\
``You asked us to disable the duplicate check in both and see what happens.
Your C rotor \emph{crashes} --- an internal sort-pointer invariant breaks ---
so line reuse there is load-bearing, not just an optimization. Ours just runs:
the model gets 1.4x bigger and the solver gives the \emph{identical} verdict.
That proves dedup only affects size and speed, never meaning. We sent you a
complete crash reproduction and a suggested fix.''}

% ---- Q&A 4: file sizes ----
\begin{frame}{Q\&A: why are the file sizes different?}
  \centering\footnotesize
  \begin{tabular}{lrr}
    \toprule
    selfie model & with comments & comments stripped\\
    \midrule
    C rotor   & 10.6 MB & 5.2 MB \quad(50.8\% was comments)\\
    Rust rotor & 3.1 MB & 3.0 MB\\
    \bottomrule
  \end{tabular}\\[0.5em]
  Same content, different spelling: comments, binary vs decimal constants,
  longer node numbers. \emph{Meaning} is identical --- proven by the solver.
\end{frame}
\note{[ANSWER --- ``shouldn't the models be the same size if they're equal?'']\\
``Size is spelling, not meaning. Half of the C file is \emph{comments} ---
strip them and 10.6 megabytes becomes 5.2. The rest is encoding: C writes
constants as 64-digit binary strings, we write decimals; its node numbers are
longer. Same model, fewer characters --- like minified versus pretty-printed
JSON. The proof they mean the same thing isn't the size; it's the solver
giving the same verdict, 36 times out of 36.''}

% ---- Q&A 5: how symbolic argv works + honesty ----
\begin{frame}{Q\&A: how symbolic argv works (and its limits)}
  \begin{itemize}\setlength\itemsep{0.35em}
    \item argument bytes $=$ \keylab{uninitialized model states} --- the
          paper's own device for ``input''; the rest of the stack is concrete.
    \item the program reads them through \emph{ordinary} loads --- decoder,
          memory, properties all untouched.
    \item \keylab{Honest scope:} argument \emph{count} is fixed; lengths
          bounded; we explore \emph{content}. Sound for C-string programs.
  \end{itemize}
\end{frame}
\note{[ANSWER --- ``how is it actually implemented, and what are the limits?'']\\
``We build the real process-start stack --- count, pointers, terminators all
concrete --- but the argument \emph{content} bytes are left uninitialized
states, which is exactly the device the paper already uses for stdin input.
The program reads them through ordinary memory loads; we changed nothing in
the decoder or the safety checks. Honest limits: we fix the \emph{number} of
arguments and bound their length, and explore their content --- which is
sound for any program that treats arguments as C strings.''}

% ---- Q&A 6: threats to validity ----
\begin{frame}{Q\&A: what we do \emph{not} claim}
  \begin{itemize}\setlength\itemsep{0.4em}
    \item Not a formal proof of generator equivalence --- that's undecidable,
          and it's the paper's own future work.
    \item Evidence is bounded: 18 programs, depth 1500, two settings.
    \item Two benchmarks agree only by \emph{both} finding nothing (UNSAT) ---
          weaker than a step-exact match.
    \item Once, an anomaly was \emph{our harness}, not the rotor --- we found
          it because one row disagreed with the other seventeen.
  \end{itemize}
\end{frame}
\note{[ANSWER --- ``so are they provably equivalent?'']\\
``No --- and we're careful to say so. A formal proof that two generators are
equivalent is undecidable; the paper itself lists it as future work. What we
have is strong \emph{checkable} evidence: 18 programs, depth 1500, two
settings, all matching. Two of those agree only by both finding nothing,
which is weaker than a step-exact hit. And once an apparent mismatch turned
out to be \emph{our own measurement harness}, not the rotor --- we caught it
precisely because one result disagreed with the other seventeen. That's the
methodology policing itself.''}
